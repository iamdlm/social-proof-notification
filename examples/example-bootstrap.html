Setup

<!-- Include Bootstrap 5 CSS -->
<link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.0/dist/css/bootstrap.min.css" rel="stylesheet">
<link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/bootstrap-icons@1.11.0/font/bootstrap-icons.css">

<!-- Include SocialProofNotification -->
<script src="../social-proof-notification.min.js"></script>

<script>
  const notifier = SocialProofNotification.create({
    theme: 'bootstrap',  // â† Set Bootstrap theme
    position: 'bottom-right',
    autoClose: true,
    autoCloseTimeout: 8000
  });
  
  notifier.init();
</script>

Custom Data Example

const notifier = SocialProofNotification.create({
  theme: 'bootstrap',
  dataSource: 'local',
  localData: [
    {
      message: '15 people purchased in the last hour!',
      timestamp: 'Just now'
    },
    {
      message: 'Sarah from California just signed up',
      timestamp: '2 minutes ago'
    }
  ]
});

notifier.init();

API Integration

Connect to your backend for real-time notifications

const notifier = SocialProofNotification.create({
  theme: 'bootstrap',
  dataSource: 'api',
  apiUrl: 'https://api.yoursite.com/notifications',
  initialDelay: 5000,
  minTimeBetween: 6  // Show every 6 hours
});

notifier.init();
