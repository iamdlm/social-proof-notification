Real-Time API Integration

Connect your backend for live notifications

// API Response Format:
// {
//   "message": "John just purchased the premium plan!",
//   "timestamp": "2 minutes ago"
// }

const notifier = SocialProofNotification.create({
  theme: 'tailwind',
  dataSource: 'api',
  apiUrl: 'https://api.yoursite.com/social-proof',
  initialDelay: 5000,
  autoCloseTimeout: 10000,
  minTimeBetween: 6  // Hours between notifications
});

notifier.init();
